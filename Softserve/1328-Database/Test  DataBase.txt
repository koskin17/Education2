-- Resorts table
CREATE TABLE Resorts (
    ResortID INT PRIMARY KEY AUTO_INCREMENT,
    ResortName VARCHAR(255) NOT NULL,
    ResortType VARCHAR(50) NOT NULL,
    QualityLevel INT NOT NULL,
    Location VARCHAR(255) NOT NULL
);

-- Clients table
CREATE TABLE Clients (
    ClientID INT PRIMARY KEY AUTO_INCREMENT,
    FirstName VARCHAR(50) NOT NULL,
    LastName VARCHAR(50) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Phone VARCHAR(20)
);

-- Agents table
CREATE TABLE Agents (
    AgentID INT PRIMARY KEY AUTO_INCREMENT,
    FirstName VARCHAR(50) NOT NULL,
    LastName VARCHAR(50) NOT NULL,
    CommissionPercentage DECIMAL(5, 2) NOT NULL
);

-- Photos table
CREATE TABLE Photos (
    PhotoID INT PRIMARY KEY AUTO_INCREMENT,
    ResortID INT NOT NULL,
    PhotoTitle VARCHAR(255) NOT NULL,
    PhotoFile VARCHAR(255) NOT NULL, -- Store file path
    PhotoTags VARCHAR(255),
    FOREIGN KEY (ResortID) REFERENCES Resorts(ResortID)
);

-- Comments table
CREATE TABLE Comments (
    CommentID INT PRIMARY KEY AUTO_INCREMENT,
    PhotoID INT NOT NULL,
    ClientID INT NOT NULL,
    CommentText TEXT NOT NULL,
    FOREIGN KEY (PhotoID) REFERENCES Photos(PhotoID),
    FOREIGN KEY (ClientID) REFERENCES Clients(ClientID)
);
